attribute vec3 point;
void main(){
	gl_PointSize=512.0;
	gl_Position=vec4(point.xyz, 1.0);
}

precision mediump float;
void main(){
	float x=gl_FragCoord.x/512.0;
	float y=gl_FragCoord.y/512.0;
	float r=0.0, g=0.0, b=0.5;

	float d=distance(vec2(0.5, 0.5), vec2(x, y));	// 0to1.4
	r=(d<0.2)?(1.0-d)*(1.0-d)*1.0:1.0;
	g=(d<0.2)?0.0:1.0;
	b=(d<0.2)?0.0:0.8;
	gl_FragColor=vec4(r, g, b, 1.0);
}
